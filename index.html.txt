    // Build lights around wheel
    function buildLights(containerId){
      const el = document.getElementById(containerId);
      const n = 18;
      for(let i=0;i<n;i++){
        const dot = document.createElement("span");
        const angle = (i / n) * Math.PI * 2;
        const r = 50; // % radius approx
        // Use container size
        // We'll position using calc from center
        dot.style.left = `calc(50% + ${Math.cos(angle) * 48}% )`;
        dot.style.top  = `calc(50% + ${Math.sin(angle) * 48}% )`;
        dot.style.transform = "translate(-50%,-50%)";
        dot.style.opacity = (i % 2 === 0) ? ".95" : ".65";
        el.appendChild(dot);
      }
    }
    buildLights("lightsA");
    buildLights("lightsB");

    // Slice centers at: 15,45,75,...,345 degrees
    const slices = [
      { label: "$1000" },
      { label: "LOSE"  },
      { label: "$100"  },
      { label: "$200"  },
      { label: "$300"  },
      { label: "LOSE"  },
      { label: "$400"  },
      { label: "$500"  },
      { label: "$600"  },
      { label: "LOSE"  },
      { label: "$700"  },
      { label: "$800"  },
    ];

    let spinning = false;
    let currentRotation = 0;

    const modalBack = document.getElementById("modalBack");
    const modalTitle = document.getElementById("modalTitle");
    const modalText  = document.getElementById("modalText");
    const claimBtn   = document.getElementById("claimBtn");
    const closeBtn   = document.getElementById("closeBtn");

    function openModal(title, text){
      modalTitle.textContent = title;
      modalText.textContent  = text;
      modalBack.style.display = "flex";
      modalBack.setAttribute("aria-hidden", "false");
    }
    function closeModal(){
      modalBack.style.display = "none";
      modalBack.setAttribute("aria-hidden", "true");
    }

    closeBtn.addEventListener("click", closeModal);
    modalBack.addEventListener("click", (e)=>{ if(e.target === modalBack) closeModal(); });
    claimBtn.addEventListener("click", ()=>{
      window.location.href = SMARTLINK_URL;
    });

    function spin(wheelElA, wheelElB){
      if(spinning) return;
      spinning = true;

      // pick random slice index
      const idx = Math.floor(Math.random() * slices.length);

      // Each slice is 30deg.
      // Pointer is at top (0deg). We want the chosen slice center to land at 0deg.
      // slice center angles: 15 + idx*30.
      const sliceCenter = 15 + idx * 30;

      // Add multiple full rotations for animation
      const extraTurns = 6; // feel real
      const target = (extraTurns * 360) - sliceCenter;

      currentRotation = currentRotation + target;

      wheelElA.style.transform = `rotate(${currentRotation}deg)`;
      wheelElB.style.transform = `rotate(${currentRotation}deg)`;

      // after animation ends
      setTimeout(()=>{
        spinning = false;

        const res = slices[idx].label;
        // ‰’ ¬„‰ („‘ Ê⁄Êœ „»«·€ ›ÌÂ«)
        openModal(
          "Spin Result",
          `You got: ${res}. Continue to confirm your entry.`
        );
      }, 4300);
    }

    const wheelA = document.getElementById("wheelA");
    const wheelB = document.getElementById("wheelB");
    document.getElementById("startA").addEventListener("click", ()=> spin(wheelA, wheelB));
    document.getElementById("startB").addEventListener("click", ()=> spin(wheelA, wheelB));
    // Optional: tap wheel to spin
    wheelA.addEventListener("click", ()=> spin(wheelA, wheelB));
    wheelB.addEventListener("click", ()=> spin(wheelA, wheelB));
